const SOLANA_RPC = 'https://api.mainnet-beta.solana.com';

export async function fetchSolanaNodes(
  limit?: number
): Promise<SolanaNode[]> {
  const response = await fetch(SOLANA_RPC, {
    method: 'POST',
    body: JSON.stringify({
      jsonrpc: '2.0',
      id: 1,
      method: 'getClusterNodes',
    }),
    next: { revalidate: 300 }, // 5min cache
  });
  
  return data.result; // ~2000+ nodes
}

⚙️ IP Extraction

function extractIP(gossip: string): string {
  // "192.168.1.1:8000" → "192.168.1.1"
  return gossip.split(':')[0];
}

🎯 maxmind.ts Core Functions

// Singleton pattern for DB reader
let reader: Reader<AsnResponse> | null;

export async function initMaxMind() {
  reader = await maxmind.open(
    'data/GeoLite2-ASN.mmdb'
  );
}

export function getASNFromMaxMind(ip: string) {
  const result = reader.get(ip);
  return {
    asn: `AS${result.autonomous_system_number}`,
    org: result.autonomous_system_organization
  };
}

export function isOVHIP(ip: string): boolean {
  const info = getASNFromMaxMind(ip);
  return OVH_ASN_LIST.includes(info?.asn);
}

OVH Filtering Pipeline

🔍 filterOVH.ts

1
Quick ASN Check
MaxMind lookup for all ~2000 nodes (<1ms each)
2
Filter OVH IPs
Check if ASN ∈ OVH_ASN_LIST (~25 matches)
3
Enrich Geolocation
ip-api.com only for OVH nodes (rate-limited)
4
Return OVHNode[]
Nodes with full IPInfo attached
export async function filterOVHNodes(
  nodes: SolanaNode[]
): Promise<OVHNode[]> {
  
  // Step 1: Quick filter with MaxMind
  for (const node of nodes) {
    const ip = extractIP(node.gossip);
    
    if (isOVHIP(ip, OVH_ASN_LIST)) {
      ovhNodes.push({
        ...node,
        ipInfo: { ip, asn, org, ... }
      });
    }
  }
  
  // Step 2: Enrich only OVH nodes
  for (const node of ovhNodes) {
    const geo = await getGeolocation(node.ip);
    node.ipInfo.country = geo.country;
    node.ipInfo.city = geo.city;
    node.ipInfo.latitude = geo.lat;
  }
  
  return ovhNodes;
}

Metrics Calculation

📊 calculateMetrics.ts

const MONTHLY_COST = 150; // EUR/node

export function calculateMetrics(
  allNodes: SolanaNode[],
  ovhNodes: OVHNode[],
  providers: Record<string, number>
): DashboardMetrics {
  
  // Market share percentage
  const marketShare = 
    (ovhNodes.length / allNodes.length) * 100;
  
  // Revenue estimation
  const revenue = 
    ovhNodes.length * MONTHLY_COST;
  
  // Geographic distribution
  const geoDistribution = {};
  ovhNodes.forEach(n => {
    const country = n.ipInfo.country_name;
    geoDistribution[country]++;
  });
  
  return { marketShare, revenue, ... };
}

Architecture Summary

🎯 Key Design Decisions

  • Offline-first ASN: MaxMind DB for instant lookups
  • Selective enrichment: Geo data only for OVH nodes
  • Cache strategy: Pre-computed JSON + stale fallback
  • Background worker: Decoupled from API requests
